<script>
let products = [
{name:"Pink Hoodie", price:"29€", vibe:"soft"},
{name:"White Top", price:"19€", vibe:"clean"},
{name:"Brown Sweater", price:"34€", vibe:"cozy"},
{name:"Baby Pink Bag", price:"24€", vibe:"soft"},
{name:"Beige Skirt", price:"27€", vibe:"clean"},
{name:"Cozy Socks", price:"12€", vibe:"cozy"}
];

let cart = [];
let currentProduct = null;

function renderProducts(list) {
let html = "";
list.forEach(p => {
html += `
<div class="card" onclick="openProduct('${p.name}','${p.price}')">
<div class="heart" onclick="event.stopPropagation(); saveItem('${p.name}')">❤️</div>
<img src="https://via.placeholder.com/300">
<div style="padding:10px">
${p.name}<br>
<b>${p.price}</b>
</div>
</div>`;
});
document.getElementById("grid").innerHTML = html;
}

function openProduct(name, price) {
currentProduct = {name, price};
document.getElementById("popup").classList.add("active");
document.getElementById("ptitle").innerText = name;
document.getElementById("pprice").innerText = price;
}

function closePopup() {
document.getElementById("popup").classList.remove("active");
}

function addToCart() {
cart.push(currentProduct);
document.getElementById("cartCount").innerText = cart.length;
}

function openCart() {
document.getElementById("cartPopup").classList.add("active");
let html = "";
cart.forEach(i => {
html += <p>${i.name} - ${i.price}</p>;
});
document.getElementById("cartItems").innerHTML = html;
}

function closeCart() {
document.getElementById("cartPopup").classList.remove("active");
}

function searchProducts(text) {
let filtered = products.filter(p => p.name.toLowerCase().includes(text.toLowerCase()));
renderProducts(filtered);
}

function filterVibe(vibe) {
let filtered = products.filter(p => p.vibe === vibe);
renderProducts(filtered);
}

function saveItem(name) {
alert(name + " saved 💖");
}

renderProducts(products);
</script>
